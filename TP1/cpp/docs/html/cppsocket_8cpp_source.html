57  struct hostent* hostinfo = NULL;
58  // gethostbyname() is obsolete!
59  if (host.empty() || !(hostinfo = ::gethostbyname(host.c_str()))) return -1; // host not found
60  ::memset(&addr, 0, sizeof(addr));
61  addr.sin_family = AF_INET;
62  addr.sin_port = htons(port);
63  // creer l'adresse du remote host a partir de son nom
64  ::memcpy((void *)&addr.sin_addr, hostinfo->h_addr_list[0], hostinfo->h_length);

257  ssize_t SocketBuffer::readLine(string& str) {
258  str.clear();
259  if (!_sock) return Socket::InvalidSocket;
260  if (!_in) _in = new InputBuffer(_inSize);
261 
262  while (true) {
263  if (retrieveLine(str, _in->remaining)) return str.length()+1;
264  // - received > 0: data received
265  // - received = 0: nothing received (shutdown or empty message)
266  // - received < 0: an error occurred

270  if (retrieveLine(str, received)) return str.length()+1;

275  bool SocketBuffer::retrieveLine(string& str, ssize_t received) {
276  if (received <= 0 || _in->begin > _in->end) {
277  _in->begin = _in->buffer;
278  return false;
279  }
280 
281  // search for separator
282  char* sep = nullptr;
283  int sepLen = 1;
284 
285  if (_inSep < 0) { // means: '\r' or '\n' or "\r\n"
286  for (char* p = _in->begin; p < _in->begin+received; ++p) {
287  if (*p == '\n') {
288  sep = p;
289  sepLen = 1;
290  break;
291  }
292  else if (*p == '\r') {
293  sep = p;
294  if (p < _in->begin+received-1 && *(p+1) == '\n') {
295  sepLen = 2;
296  }
297  break;
298  }
299  }
300  }
301  else {
302  for (char* p = _in->begin; p < _in->begin+received; ++p)
303  if (*p == _inSep) {
304  sep = p;
305  break;
306  }
307  }
308 
309  if (sep) {
310  str.append(_in->begin, sep-_in->begin);
311  _in->remaining = received - (sep+sepLen - _in->begin);
312  _in->begin = sep+sepLen;
313  return true;
314  }
315  else {
316  str.append(_in->begin, received);
317  _in->begin = _in->buffer;
318  _in->remaining = 0;
319  return false;
320  }
321  }

324  ssize_t SocketBuffer::writeLine(const string& str) {
325  if (!_sock) return Socket::InvalidSocket;
326 
327  // a negature value of _outSep means that \r\n must be added
328  size_t len = str.length() + (_outSep < 0 ? 2 : 1);
329 
330  // if len is not too large, try to send everything in one block
331  if (len <= _outSize) {
332  char buffer[len];
333  ::memcpy(buffer, str.c_str(), str.length()); // copy string
334  if (_outSep >= 0) buffer[len-1] = _outSep; // add separator
335  else {
336  buffer[len-2] = '\r';
337  buffer[len-1] = '\n';
338  }
339  return write(buffer, len);
340  }
341  else {
342  ssize_t sent = write(str.c_str(),str.length());
343  if (_outSep >= 0)
344  sent += _sock->send(&_outSep, 1);
345  else
346  sent+= _sock->send("\r\n", 2);
347  return sent;
348  }
349  }

352  ssize_t SocketBuffer::write(const char* s, size_t len) {
353  if (!_sock) return Socket::InvalidSocket;
354  const char* begin = s;
355  const char* end = s + len;
356  ssize_t totalSent = 0;
357 
358  while (begin < end) {
359  // - sent > 0: data sent
360  // - sent = 0: file was shutdown
361  // - sent < 0: an error occurred
362  ssize_t sent = _sock->send(begin, end-begin);
363  if (sent <= 0) return sent; // -1 (error) or 0 (shutdown)
364  begin += sent;
365  totalSent += sent;
366  }
367  return totalSent;
368  }
